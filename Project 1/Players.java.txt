package assignment3_000924316;
/**
 * Implementation of a measuring Scale with a random weight on the
 * left side. The user can manipulate the right to try and balance it.
 * @author vanshul vanshul
 */

import javafx.scene.canvas.GraphicsContext;
import javafx.scene.paint.Color;

import java.util.Random;

public class Players {
    private double x;
    private double y;
    private int average;
    private Color color;
    private String title;
    private     Head head;
    private Wheel wheel1,wheel2;
    
    public Players(double x, double y, Color color){
        this.x=x;
        this.y=y;
        this.color=color;
        this.title="";
        
        Random rand= new Random();
        this.average=rand.nextInt(100)+1;
        Head head = new Head(x, y);
        wheel1=new Wheel(x,y+38);
        wheel2=new Wheel(x+73,y+38);

    }
    public  Players(double x, double y,Color color,String title){
        this.x=x;
        this.y=y;
        this.color=color;
        this.title=title;
        this.average=100;
        Head head= new Head(x,y);
        wheel1=new Wheel(x,y+38);
        wheel2=new Wheel(x+73,y+38);
    }
    public int getAverage(){
        return this.average;
    }
    public void draw(GraphicsContext gc) {
        //drawing player with team color
        gc.setFill(color);
        //drawing player body
        gc.fillOval(x+7.5, y+28+10, 65, 65);
        gc.setFill(Color.BLACK);
        //drawing player score
        gc.fillText(average+"", x+32.5, y+38+32.5);
        //drawing player text
        gc.fillText(title, x,y+38+65+10);
        //drawing player's head and side wheels
        Head.draw(gc);
        wheel1.draw(gc);
        wheel2.draw(gc);
    }
    
}
